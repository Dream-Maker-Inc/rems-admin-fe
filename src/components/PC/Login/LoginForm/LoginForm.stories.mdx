  const [{}, updateArgs] = useArgs();
  const handleIdChange = (v) => updateArgs({ idValue: v });
  const handlePwChange = (v) => updateArgs({ pwValue: v });
  const handleRememberChange = (v) => updateArgs({ rememberMe: v });
  return (
    <LoginForm
      handleIdChange={handleIdChange}
      handlePwChange={handlePwChange}
      handleRememberChange={handleRememberChange}
      handleLoginSubmit={() => alert("login ok")}
      {...args}
    />
  );
};
